namespace Csharp_Parte2;

class Program
{
    static void Main(string[] args)
    {
        Apresentacao();
        string tipo = pegarTipo();
        int qtd = qtdIngresso();
        string resultado = Resu(tipo, qtd);
        Console.WriteLine(resultado);
    }

    public static void Apresentacao()
    {
        Console.WriteLine("PROGRAMA DO INGRESSO");
    }

    public static string pegarTipo()
    {
        Console.WriteLine("Informe o tipo do ingresso (Meia ou Inteira):");
        string tipo = Console.ReadLine().ToLower();
        return tipo;
    }

    public static int qtdIngresso()
    {
        Console.WriteLine("Informe a quantidade de ingressos:");
        int qtd = Convert.ToInt32(Console.ReadLine());
        return qtd;
    }

    public static string Resu(string tipo, int qtd)
    {
        double precoUnitario = tipo == "meia" ? 14.25 : 28.50;
        double total = 0;

        for (int i = 0; i < qtd; i++)
        {
            total += precoUnitario;
        }

        return $"Total a pagar: R$ {total}";
    }
}

class Program
{
    static void Main(string[] args)
    {
        Console.WriteLine("Informe a cor do semáforo:");
        string cor = Console.ReadLine().ToLower();
        string acao = acaoFarol(cor);
        Console.WriteLine(acao);
    }

    static string acaoFarol(string cor)
    {
        var coresAcoes = new (string cor, string acao)[]
        {
            ("verde", "atravesse"),
            ("vermelho", "espere")
        };

        for (int i = 0; i < coresAcoes.Length; i++)
        {
            if (coresAcoes[i].cor == cor)
            {
                return coresAcoes[i].acao;
            }
        }

        return "Farol inoperante";
    }
}

    }

    public static void Apresentacao()
    {
        Console.WriteLine("PROGRAMA DO PESO SORVETE ");
    }

    public static double pegarPeso()
    {
        Console.WriteLine("Informe o peso do sorvete");
        double peso = Convert.ToDouble(Console.ReadLine());
        return peso;
    }

    public static void Valor(double peso)
    {
        if (peso <= 0)
        {
            Console.WriteLine("Peso Inválido");
            return;
        }

        double precoPor100g = (peso >= 1000) ? 3.00 : 3.50;
        double total = (peso / 100) * precoPor100g;
        Console.WriteLine($"O total a se pagar é: R$ {total}");
    }
}

    {

    }

    public static void Valor(double temp)
    {
        if (temp >= 41)
        {
            Console.WriteLine("A situação para a sua temperatura é Hipertamia");
        }
        else if (temp >= 39.6 && temp < 41)
        {
            Console.WriteLine("A situação para a sua temperatura é Febre Alta");
        }
        else if (temp >= 37.6 && temp < 39.6)
        {
            Console.WriteLine("A situação para a sua temperatura é Febre");
        }
        else if (temp >= 36 && temp < 37.6)
        {
            Console.WriteLine("A situação para a sua temperatura é Normal");
        }
        else if (temp < 36)
        {
            Console.WriteLine("A situação para a sua temperatura é Hipotermia");
        }

    }
}



using System;

namespace Csharp_Parte2;

class Program
{
    static void Main(string[] args)
    {
        Apresentacao();
        double nota1 = ObterNota("primeira");
        double nota2 = ObterNota("segunda");
        double nota3 = ObterNota("terceira");
        double media = CalcularMedia(nota1, nota2, nota3);
        string situacao = AvaliarSituacao(media);
        Console.WriteLine($"Média: {media:F2}");
        Console.WriteLine($"Situação: {situacao}");
    }

    public static void Apresentacao()
    {
        Console.WriteLine("PROGRAMA DE CÁLCULO DE MÉDIA ESCOLAR");
    }

    public static double ObterNota(string ordem)
    {
        Console.Write($"Informe a {ordem} nota: ");
        return Convert.ToDouble(Console.ReadLine());
    }

    public static double CalcularMedia(double n1, double n2, double n3)
    {
        return (n1 + n2 + n3) / 3;
    }

    public static string AvaliarSituacao(double media)
    {
        if (media >= 8)
        {
            return "Aprovado com sucesso";
        }
        else if (media >= 6 && media < 8)
        {
            return "Aprovado";
        }
        else if (media >= 3 && media < 6)
        {
            return "recuperação";
        }
        else if (media < 3)
        {
            return "Reprovado";
        }
        else
        {
            return "Desistente";
        }
    }
}



using System;

namespace Csharp_Parte2;

class Program
{
    static void Main(string[] args)
    {
        Apresentacao();
        double altura = ObterAltura();
        double peso = ObterPeso();
        double imc = CalcularIMC(peso, altura);
        string situacao = AvaliarSituacao(imc);
        Console.WriteLine($"O IMC é {imc:F2} e a situação é {situacao}");
    }

    public static void Apresentacao()
    {
        Console.WriteLine("PROGRAMA DE CÁLCULO DE IMC");
    }

    public static double ObterAltura()
    {
        Console.Write("Informe a altura (em metros): ");
        return Convert.ToDouble(Console.ReadLine());
    }

    public static double ObterPeso()
    {
        Console.Write("Informe o peso (em kg): ");
        return Convert.ToDouble(Console.ReadLine());
    }

    public static double CalcularIMC(double peso, double altura)
    {
        return peso / (altura * altura);
    }

    public static string AvaliarSituacao(double imc)
    {
        if (imc < 18.5)
        {
            return "Abaixo do peso";
        }
        else if (imc >= 18.5 && imc < 25)
        {
            return "Peso normal";
        }
        else if (imc >= 25 && imc < 30)
        {
            return "Sobrepeso";
        }
        else if (imc >= 30 && imc < 35)
        {
            return "Obesidade grau I";
        }
        else if (imc >= 35 && imc < 40)
        {
            return "Obesidade grau II";
        }
        else
        {
            return "Obesidade grau III";
        }
    }
}
